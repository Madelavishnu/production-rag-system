3️⃣ Reverse a Number


#!/bin/bash

echo "Enter a number:"
read num

rev=0

while [ $num -gt 0 ]
do
    rem=$((num % 10))
    rev=$((rev * 10 + rem))
    num=$((num / 10))
done

echo "Reversed number is: $rev"

9️⃣ C Program – Avoid Zombie using wait()


#include       // printf
#include      // exit
#include      // fork, getpid
#include   // pid_t
#include    // wait, WEXITSTATUS

int main()
{
    pid_t pid;
    int status;

    pid = fork();

    if (pid < 0) {
        printf("Fork failed\n");
        exit(1);
    }

    if (pid > 0) {
        // Parent process
        printf("I'm parent with PID %d\n", getpid());
        wait(&status);   // Parent waits for child
        printf("Child terminated with status %d\n", WEXITSTATUS(status));
    }
    else {
        // Child process
        printf("I'm child with PID %d\n", getpid());
        exit(2);
    }

    return 0;
}
